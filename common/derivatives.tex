% partial derivatives
\newcommand{\pd}[2]       { \frac{\partial #1}     {\partial #2} }
\newcommand{\pdt}[1]      { \frac{\partial #1}     {\partial t} }
\newcommand{\pds}[2]      { \frac{\partial #1^{2}} {\partial #2^{2}} }

% total derivatives
\newcommand{\td}[2]       { \frac{d #1}     {d #2} }
\newcommand{\tdt}[1]      { \frac{d #1}     {d t} }
\newcommand{\tds}[2]      { \frac{d^{2} #1} {d #2^{2}} }

% del operators
\newcommand{\deldot}[0]   { \nabla \cdot }
\newcommand{\delcross}[0] { \nabla \times }
